مفهوم: namespace ها نقش scope برای برای توابع و کلاس‌ها را دارند و به ما اجازه می‌دهند تا در یه پروژه از چندین کلاس یا تابع با اسم یکسان استفاده کنیم.

Conflict
function output(){ echo "Rendered HTML!"; } function output(){ echo "RSS"; }
Without Conflict
namespace HTML; function output(){ echo "Rendered HTML!"; } namsepace RSS; function output(){ echo "RSS"; } \Rss\output();// RSS \HTML\output();// Rendered HTML!
file1.php
namespace MyProject; public class Hey{ public function sayHello(){ echo "Hi"; } }
file2.php
require('file1.php'); $hey = \MyProject\Hey(); $hey->sayHello();// Hi
